52auto shortlex_compare(const std::string& lhs, const std::string& rhs) {
53 auto comp = lhs.length() <=> rhs.length();
54
55 if (comp == std::strong_ordering::equal) {
56 return lhs <=> rhs;
57 } else {
58 return comp;
59 }
60}

63struct Rule {
65 std::string left;
67 std::string right;
68
69 Rule() = delete;
70 Rule(std::string left, std::string right) : left(std::move(left)), right(std::move(right)) {};
71
74 int order() {
75 if (shortlex_compare(this->left, this->right) == std::strong_ordering::less) {
76 std::swap(this->left, this->right);
77 return 1;
78 }
79
80 return 0;
81 }
82};

91template<class L>
92concept Lookup = requires(
93 L l,
94 const std::vector<Rule>& rules,
95 const std::vector<bool>& active,
96 const int& n_active,
97 const Rule& rule,
98 size_t idx,
99 std::string& string
100) {
101 { L(rules, active, n_active, 100) } -> std::same_as<L>;
104 l.insert(rule, idx);
106 l.remove(rule);
108 l.rewrite(string);
109};

137 void add_rule(Rule&& rule) {
138 this->active.push_back(true);
139 this->n_active += 1;
140 this->rules.push_back(std::move(rule));
141
142 this->lookup.insert(this->rules[rules.size() - 1], rules.size() - 1);
143 }
144
146 void remove_rule(Rule&& to_check, int i) {
147 this->active[i] = false;
148 this->n_active -= 1;
149 this->stack.push_back(to_check);
150
151 this->lookup.remove(to_check);
152 }
153
156 void clear_stack() {
157 while (!this->stack.empty()) {
158 Rule rule = std::move(this->stack.back());
159 this->stack.pop_back();
160 this->lookup.rewrite(rule.left);
161 this->lookup.rewrite(rule.right);
162
163 if (rule.left != rule.right) {
164 rule.order();
165 this->add_rule(std::move(rule));
166 Rule& rule_ = this->rules.back();
167
168 for (size_t i = 0; i < this->rules.size() - 1; i++) {
169 if (this->is_active(i)) {
170 Rule& to_check = this->rules[i];
171 if (to_check.left.find(rule_.left) != std::string::npos) {
172 this->remove_rule(std::move(to_check), i);
173 } else if (to_check.right.find(rule_.left) != std::string::npos) {
174 this->lookup.rewrite(to_check.right);
175 }
176 }
177 }
178 }
179 }
180 }

185 void overlap(size_t i, size_t j) {
186 Rule& a = this->rules[i];
187 Rule& b = this->rules[j];
188
189 size_t m = std::min(a.left.length(), b.left.length());
190
191 for (size_t k = 1; k < m; k++) {
192 // Refresh references.
193 Rule& a = this->rules[i];
194 Rule& b = this->rules[j];
195 std::string_view suffix {a.left.data() + (a.left.length() - k), k};
196
197 if (b.left.starts_with(suffix)) {
198 // a.left = XY, b.left = YZ
199 // xbr = X|b.right, arz = a.right|Z
200 std::string xbr;
201 xbr.reserve(a.left.length() - k + b.right.length());
202 xbr.append(a.left, 0, a.left.length() - k);
203 xbr.append(b.right);
204
205 std::string arz;
206 arz.reserve(a.right.length() + b.left.length() - k);
207 arz.append(a.right);
208 arz.append(b.left, k, b.left.length() - k);
209
210 this->stack.emplace_back(xbr, arz);
211 this->clear_stack();
212
213 if (!(this->is_active(i) && this->is_active(j))) {
214 return;
215 }
216 }
217 }
218 }

220 KnuthBendixState(const std::vector<Rule>& initial_rules)
221 : lookup(rules, active, n_active, initial_rules.size())
222 {
223 this->stack.reserve(initial_rules.size());
224 this->rules.reserve(initial_rules.size());
225 this->active.reserve(initial_rules.size());
226 this->n_active = 0;
227
228 for (const Rule& rule : initial_rules) {
229 this->stack.push_back(rule);
230 this->clear_stack();
231 }
232 }
233};

237template<Lookup L>
238std::vector<Rule> knuthbendix(const std::vector<Rule>& initial_rules) {
239 // Setup
240 KnuthBendixState<L> state {initial_rules};
241
242 // Check overlaps
243 for (size_t i = 0; i < state.rules.size(); i++) {
244 #ifndef BENCHMARK
245 printf("\rActive/Generated Rules: %d/%zu", state.n_active, state.rules.size());
246 fflush(stdout);
247 #endif
248 for (size_t j = 0; j <= i && state.is_active(i); j++) {
249 if (state.is_active(j)) state.overlap(i, j);
250 if (j < i && state.is_active(j) && state.is_active(i)) {
251 state.overlap(j, i);
252 }
253 }
254 }
255
256 #ifndef BENCHMARK
257 printf("\n");
258 #endif
259
260 std::vector<Rule> confluent;
261 confluent.reserve(state.n_active);
262
263 for (size_t i = 0; i < state.rules.size(); i++) {
264 if (state.is_active(i)) {
265 confluent.push_back(std::move(state.rules[i]));
266 }
267 }
268
269 return confluent;
270}
